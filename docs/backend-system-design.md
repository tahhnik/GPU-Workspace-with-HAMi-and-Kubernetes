### Workspace Pod YAML Template

```yaml
apiVersion: v1
kind: Pod
metadata:
  name: workspace
  namespace: user-${USER_ID}
  labels:
    app: workspace
    user: ${USER_ID}
spec:
  runtimeClassName: nvidia
  schedulerName: hami-scheduler
  automountServiceAccountToken: false

  securityContext:
    runAsNonRoot: true
    runAsUser: 1000
    runAsGroup: 1000
    fsGroup: 1000

  containers:
  - name: workspace
    image: registry.ai.poridhi.io/workspace:latest
    ports:
    - name: jupyter
      containerPort: 8888
    - name: vscode
      containerPort: 8080

    env:
    - name: JUPYTER_TOKEN
      valueFrom:
        secretKeyRef:
          name: workspace-secrets
          key: jupyter-token

    resources:
      requests:
        nvidia.com/gpu: 1
        nvidia.com/gpumem: ${VRAM_MB}
        cpu: "1"
        memory: "4Gi"
      limits:
        nvidia.com/gpu: 1
        nvidia.com/gpumem: ${VRAM_MB}
        cpu: "${CPU_LIMIT}"
        memory: "${MEMORY_LIMIT}"

    securityContext:
      allowPrivilegeEscalation: false
      capabilities:
        drop: ["ALL"]

    volumeMounts:
    - name: home
      mountPath: /home/user
    - name: dshm
      mountPath: /dev/shm

    livenessProbe:
      httpGet:
        path: /api/status
        port: 8888
      initialDelaySeconds: 30
      periodSeconds: 30

    readinessProbe:
      httpGet:
        path: /api/status
        port: 8888
      initialDelaySeconds: 10
      periodSeconds: 5

  volumes:
  - name: home
    persistentVolumeClaim:
      claimName: home
  - name: dshm
    emptyDir:
      medium: Memory
      sizeLimit: "2Gi"

  restartPolicy: Never
```

## Ingress & URL Routing

### Option 1: Subdomain per User (Recommended)

```
https://alice.ai.poridhi.io/     вҶ’ user-alice namespace, JupyterLab
https://bob.ai.poridhi.io/       вҶ’ user-bob namespace, JupyterLab
https://carol.ai.poridhi.io/     вҶ’ user-carol namespace, JupyterLab
```

**Requirements:**
- Wildcard DNS: `*.ai.poridhi.io вҶ’ Load Balancer IP`
- Wildcard TLS certificate (Let's Encrypt or purchased)

### Option 2: Path-based Routing (Simpler)

```
https://ai.poridhi.io/user/alice/lab    вҶ’ user-alice, JupyterLab
https://ai.poridhi.io/user/alice/code   вҶ’ user-alice, VS Code
https://ai.poridhi.io/user/bob/lab      вҶ’ user-bob, JupyterLab
```

**Requirements:**
- Single domain DNS
- Single TLS certificate

### Ingress YAML (Path-based)

```yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: workspace-ingress
  namespace: user-alice
  annotations:
    nginx.ingress.kubernetes.io/rewrite-target: /$2
    nginx.ingress.kubernetes.io/proxy-body-size: "0"
    nginx.ingress.kubernetes.io/proxy-read-timeout: "3600"
    nginx.ingress.kubernetes.io/proxy-send-timeout: "3600"
    nginx.ingress.kubernetes.io/websocket-services: "workspace"
spec:
  ingressClassName: nginx
  tls:
  - hosts:
    - ai.poridhi.io
    secretName: ai-poridhi-tls
  rules:
  - host: ai.poridhi.io
    http:
      paths:
      - path: /user/alice(/|$)(.*)
        pathType: Prefix
        backend:
          service:
            name: workspace
            port:
              number: 8888
```

## Database Schema

### PostgreSQL Schema

```sql
-- Users table
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email           VARCHAR(255) UNIQUE NOT NULL,
    password_hash   VARCHAR(255) NOT NULL,
    tier            VARCHAR(50) DEFAULT 'free',
    namespace       VARCHAR(63) UNIQUE,
    created_at      TIMESTAMP DEFAULT NOW(),
    last_login      TIMESTAMP,
    is_active       BOOLEAN DEFAULT true,
    email_verified  BOOLEAN DEFAULT false
);

-- Sessions (workspace instances)
CREATE TABLE sessions (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE CASCADE,
    pod_name        VARCHAR(63),
    status          VARCHAR(50),  -- starting, running, stopping, stopped, error
    started_at      TIMESTAMP,
    stopped_at      TIMESTAMP,
    gpu_memory_mb   INTEGER,
    ide_type        VARCHAR(50),  -- jupyterlab, vscode
    created_at      TIMESTAMP DEFAULT NOW()
);

-- Usage logs for billing/monitoring
CREATE TABLE usage_logs (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE CASCADE,
    session_id      UUID REFERENCES sessions(id) ON DELETE SET NULL,
    date            DATE NOT NULL,
    gpu_seconds     INTEGER DEFAULT 0,
    gpu_memory_mb   INTEGER,
    created_at      TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, date)
);

-- Tier definitions
CREATE TABLE tiers (
    name                VARCHAR(50) PRIMARY KEY,
    gpu_memory_mb       INTEGER NOT NULL,
    cpu_cores           INTEGER NOT NULL,
    ram_mb              INTEGER NOT NULL,
    storage_gb          INTEGER NOT NULL,
    max_session_hours   INTEGER NOT NULL,
    monthly_gpu_hours   INTEGER,  -- NULL = unlimited
    price_monthly       DECIMAL(10,2)
);

-- Insert default tiers
INSERT INTO tiers VALUES
    ('free',      4096,  2,  8192,  20, 2,   20,    0.00),
    ('standard',  8192,  4, 16384,  50, 4,  100,   29.99),
    ('premium',  16384,  8, 32768, 100, 8, NULL,   99.99);

-- Refresh tokens
CREATE TABLE refresh_tokens (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id         UUID REFERENCES users(id) ON DELETE CASCADE,
    token_hash      VARCHAR(255) NOT NULL,
    expires_at      TIMESTAMP NOT NULL,
    created_at      TIMESTAMP DEFAULT NOW()
);

-- Indexes
CREATE INDEX idx_sessions_user_id ON sessions(user_id);
CREATE INDEX idx_sessions_status ON sessions(status);
CREATE INDEX idx_usage_logs_user_date ON usage_logs(user_id, date);
CREATE INDEX idx_refresh_tokens_user ON refresh_tokens(user_id);
```

### Workspace Image

```dockerfile
# Base: NVIDIA PyTorch image
FROM nvcr.io/nvidia/pytorch:24.01-py3

# Install JupyterLab
RUN pip install jupyterlab jupyterlab-git

# Install code-server (VS Code)
RUN curl -fsSL https://code-server.dev/install.sh | sh

# Install common ML packages
RUN pip install \
    transformers \
    datasets \
    accelerate \
    wandb \
    tensorboard \
    scikit-learn \
    pandas \
    matplotlib \
    seaborn

# Create non-root user
RUN useradd -m -u 1000 user
USER user
WORKDIR /home/user

# Expose ports
EXPOSE 8888 8080

# Start script
COPY --chown=user:user start.sh /start.sh
CMD ["/start.sh"]
```
